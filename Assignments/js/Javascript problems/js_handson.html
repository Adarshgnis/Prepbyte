        {

            const alphabets = []
            // with the help of this loop we converted the string s into an array
            for(let i=0; i < s.length; i++){
              alphabets.push(s.charAt(i) );
            }
            // We filtered the repetative value and store in the variable repeat_value 
            const repeat_value = alphabets.filter( (alphabet, index) => alphabets.indexOf(alphabet) !==  alphabets.lastIndexOf(alphabet) )
            // then we sorted this array
            repeat_value.sort(repeat_value)

            let newStr = ''

            counter = 1

                counter += 1

You are given a string S, Your task is to check wether the given string is a Palindrome or not. Palindrome is a string, which turnout same when read in reverse direction. Example: "naman" is a Palindrome. String can contain both upppercase lowercase letters. 


        var Palin_Check = (str) =>
        {
           const forwardArr = []
           const backwardArr = []
           let counter = 0
           
           for(let i=0; i < str.length; i++) {
             forwardArr.push(str.charAt(i))
           }
           
           for(let i = str.length-1; i >= 0; i--) {
             backwardArr.push(str.charAt(i))
           }
           
           for(let i = 0; i < str.length; i++) {
             if(forwardArr[i] === backwardArr[i]) {
               counter += 1
             }
           }
           
           if(counter === str.length) {
             return "True"
           } else {
             return "False"
           }
        }
